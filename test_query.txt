create table students (
id char(10) not null,
name char(20),
primary key(id)
);

create table lectures (
id int not null,
name char(20),
capacity int,
primary key(id)
);

create table ref(
id int,
foreign key (id) references lectures (id)
);

create table apply(
s_id char(10) not null,
I_id int not null,
apply_date date,
primary key (s_id, I_id),
foreign key (s_id) references students(id),
foreign key(I_id) references lectures(id)
);
